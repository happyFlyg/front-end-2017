json-loader：cmd的方式

  1. cnpm install json-loader --save-dev
  2. index.js里面： 
    		   	define(["./stu.json"],function(stu_json){
		   		console.log(stu_json.name);
		   		return document.getElementById("test").innerHTML=txt;
		   	})
  3. config.js文件里面： 
    				module: {
					loaders: [{
						test: /\.json$/,
						loader: "json"
					}]
				}
  4. webpack ./src/js/index.js ./assets/js/index.js
  5. cnpm start
